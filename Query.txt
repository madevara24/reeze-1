View project member (siapa saja yang join project ini)
SELECT users.username as 'username', projects.name as 'project name' FROM project_members JOIN users ON project_members.user_id = users.id JOIN projects ON project_members.project_id = projects.id WHERE project_members.project_id = 3

View your projects (project apa saja yang anda kerjakan)
SELECT users.username as 'username', projects.name as 'project name' FROM project_members JOIN users ON project_members.user_id = users.id JOIN projects ON project_members.project_id = projects.id WHERE project_members.user_id = 1

view cards by project, iteration
SELECT cards.id, projects.name as 'project', iteration, req.username as 'requester', own.username as 'owner', type, cards.description as 'description', points, github_branch_name as 'github branch' FROM cards JOIN users req on cards.requester = req.id join users own on cards.owner = own.id JOIN projects on cards.project_id = projects.id WHERE cards.project_id = 1 ORDER BY cards.iteration, cards.points ASC
